Render in Controller 
<?php

namespace App\Controller;

use Symfony\Bundle\FrameworkBundle\Controller\Controller;

final class SomeController extends Controller
{
    public function listAction()
    {
        // renders templates/some/list.html.twig
        return $this->render('some/list.html.twig');
    }
}

<?php

namespace App\Presenter;

use Nette\Application\UI\Presenter;

final class SomePresenter extends Presenter
{
    public function actionList()
    {
        // app/templates/Some/list.latte is rendered by default,
        // where "Some" is presenter prefix and "list" is action suffix
    }
}
Render Variable in controller 
<?php

namespace App\Controller;

use Symfony\Bundle\FrameworkBundle\Controller\Controller;

final class SomeController extends Controller
{
    public function listAction()
    {
        return $this->render('some/list.html.twig', [
            'someKey' => 'someValue'
        ]);
    }
}

<?php

namespace App\Presenter;

use Nette\Application\UI\Presenter;

final class SomePresenter extends Presenter
{
    public function actionList()
    {
        $this->template->add('someKey', 'someValue');
    }
}
Get Service in Controller 
<?php

namespace App\Controller;

use App\SomeService;

final class SomeController
{
    /**
     * @var SomeService
     */
    private $someService;

    public function __construct(SomeService $someService)
    {
        $this->someService = $someService;
    }

    public function listAction()
    {
        $someResult = $this->someService->someMethod();
    }
}

<?php

namespace App\Presenter;

use App\SomeService;
use Nette\Application\UI\Presenter;

final class SomePresenter extends Presenter
{
    /**
     * @var SomeService
     */
    private $someService;

    public function __construct(SomeService $someService)
    {
        $this->someService = $someService;
    }

    public function actionList()
    {
        $someResult = $this->someService->someMethod();
    }
}
Invokeable controller 
<?php

namespace App\Controller;

use Symfony\Component\HttpFoundation\Response;

final class SomeController
{
    public function __invoke(): Response
    {
        return new Response('some content');
    }
}

<?php

namespace App\Presenter;

use Nette\Application\IPresenter;
use Nette\Application\IResponse;
use Nette\Application\Responses\TextResponse;

final class SomePresenter implements IPresenter
{
    public function run(Request $request): IResponse
    {
        return new TextResponse('some content');
    }
}
